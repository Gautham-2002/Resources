# ============================================
# FILE: scripts/create-topics.sh
# Create example topics
# ============================================

#!/bin/bash

echo "Creating Kafka Topics"
echo "===================="
echo ""

BROKER="localhost:9092"

# Check if Kafka is running
if ! docker ps | grep -q kafka; then
    echo "❌ Kafka is not running. Start it with:"
    echo "   docker-compose -f docker-compose-single.yml up -d"
    exit 1
fi

# Function to create topic
create_topic() {
    local topic=$1
    local partitions=$2
    local replication=$3
    
    echo "Creating topic: $topic (partitions=$partitions, replication=$replication)"
    
    docker exec -it kafka kafka-topics \
        --bootstrap-server $BROKER \
        --create \
        --topic $topic \
        --partitions $partitions \
        --replication-factor $replication \
        --if-not-exists 2>/dev/null
    
    if [ $? -eq 0 ]; then
        echo "✓ Topic $topic created"
    else
        echo "✗ Failed to create $topic"
    fi
}

# Create topics
create_topic "my-first-topic" 3 1
create_topic "go-topic" 3 1
create_topic "orders" 6 1
create_topic "orders-dlq" 1 1
create_topic "payments" 3 1
create_topic "events" 6 1
create_topic "perf-test" 6 1

echo ""
echo "Listing all topics:"
docker exec -it kafka kafka-topics --bootstrap-server $BROKER --list

echo ""
echo "✓ Topics created successfully!"

# ============================================
# FILE: scripts/reset-offsets.sh
# Reset consumer group offsets
# ============================================

#!/bin/bash

BROKER="localhost:9092"
GROUP="${1}"
TOPIC="${2}"
RESET_TO="${3:-earliest}"

if [ -z "$GROUP" ] || [ -z "$TOPIC" ]; then
    echo "Usage: ./reset-offsets.sh <consumer-group> <topic> [earliest|latest|offset]"
    echo ""
    echo "Examples:"
    echo "  ./reset-offsets.sh my-group my-topic earliest"
    echo "  ./reset-offsets.sh my-group my-topic latest"
    exit 1
fi

echo "Resetting offsets for group: $GROUP, topic: $TOPIC"
echo "Reset to: $RESET_TO"
echo ""

# Show current offsets
echo "Current offsets:"
docker exec kafka kafka-consumer-groups \
    --bootstrap-server $BROKER \
    --group $GROUP \
    --describe

echo ""
read -p "Continue with reset? [y/N] " confirm

if [ "$confirm" != "y" ]; then
    echo "Cancelled"
    exit 0
fi

# Reset offsets
docker exec kafka kafka-consumer-groups \
    --bootstrap-server $BROKER \
    --group $GROUP \
    --topic $TOPIC \
    --reset-offsets \
    --to-$RESET_TO \
    --execute

echo ""
echo "✓ Offsets reset to $RESET_TO"

# Show new offsets
echo ""
echo "New offsets:"
docker exec kafka kafka-consumer-groups \
    --bootstrap-server $BROKER \
    --group $GROUP \
    --describe

# ============================================
# FILE: scripts/produce-json.sh
# Produce JSON messages
# ============================================

#!/bin/bash

BROKER="localhost:9092"
TOPIC="${1:-orders}"
COUNT="${2:-20}"

echo "Producing JSON messages to: $TOPIC"
echo "==================================="
echo ""

for i in $(seq 1 $COUNT); do
    order_id="ORD-$(printf "%05d" $i)"
    amount=$(awk -v min=10 -v max=500 'BEGIN{srand(); print min+rand()*(max-min)}')
    timestamp=$(date -u +"%Y-%m-%dT%H:%M:%SZ")
    
    json="{\"orderId\":\"$order_id\",\"amount\":$amount,\"timestamp\":\"$timestamp\"}"
    
    echo $json | docker exec -i kafka kafka-console-producer \
        --bootstrap-server $BROKER \
        --topic $TOPIC \
        --property "parse.key=true" \
        --property "key.separator=:" <<< "$order_id:$json" 2>/dev/null
    
    echo "✓ Produced: $json"
    sleep 0.2
done

echo ""
echo "✓ Produced $COUNT JSON messages to $TOPIC"

# ============================================
# FILE: scripts/performance-test.sh
# Run performance tests
# ============================================

#!/bin/bash

BROKER="localhost:9092"
TOPIC="perf-test"
NUM_RECORDS=100000
RECORD_SIZE=1024
THROUGHPUT=-1

echo "Kafka Performance Test"
echo "======================"
echo ""
echo "Topic: $TOPIC"
echo "Records: $NUM_RECORDS"
echo "Record Size: $RECORD_SIZE bytes"
echo "Target Throughput: unlimited"
echo ""

# Create topic if not exists
docker exec kafka kafka-topics \
    --bootstrap-server $BROKER \
    --create \
    --topic $TOPIC \
    --partitions 6 \
    --replication-factor 1 \
    --if-not-exists 2>/dev/null

echo "Running producer performance test..."
echo "---"

docker exec kafka kafka-producer-perf-test \
    --topic $TOPIC \
    --num-records $NUM_RECORDS \
    --record-size $RECORD_SIZE \
    --throughput $THROUGHPUT \
    --producer-props bootstrap.servers=$BROKER \
    acks=1 \
    compression.type=lz4

echo ""
echo "---"
echo "Running consumer performance test..."
echo "---"

docker exec kafka kafka-consumer-perf-test \
    --bootstrap-server $BROKER \
    --topic $TOPIC \
    --messages $NUM_RECORDS \
    --threads 4 \
    --group perf-test-group

echo ""
echo "✓ Performance test completed"

# ============================================
# FILE: scripts/broker-health.sh
# Check broker health
# ============================================

#!/bin/bash

BROKER="localhost:9092"

echo "Kafka Broker Health Check"
echo "=========================="
echo ""

# Check if containers are running
echo "Container Status:"
echo "---"
docker ps --filter "name=kafka" --format "table {{.Names}}\t{{.Status}}\t{{.Ports}}"
echo ""

# Check broker API versions
echo "Broker API Versions:"
echo "---"
docker exec kafka kafka-broker-api-versions --bootstrap-server $BROKER 2>/dev/null | head -5
echo ""

# Check under-replicated partitions
echo "Under-Replicated Partitions:"
echo "---"
result=$(docker exec kafka kafka-topics --bootstrap-server $BROKER --describe --under-replicated-partitions 2>/dev/null)
if [ -z "$result" ]; then
    echo "✓ No under-replicated partitions"
else
    echo "$result"
fi
echo ""

# Check unavailable partitions
echo "Unavailable Partitions:"
echo "---"
result=$(docker exec kafka kafka-topics --bootstrap-server $BROKER --describe --unavailable-partitions 2>/dev/null)
if [ -z "$result" ]; then
    echo "✓ No unavailable partitions"
else
    echo "$result"
fi
echo ""

# Disk usage
echo "Disk Usage:"
echo "---"
docker exec kafka du -sh /var/lib/kafka/data 2>/dev/null
echo ""

# Active consumer groups
echo "Active Consumer Groups:"
echo "---"
docker exec kafka kafka-consumer-groups --bootstrap-server $BROKER --list 2>/dev/null
echo ""

echo "✓ Health check completed"

#!/bin/bash

BROKER="localhost:9092"
TOPIC="quick-test"

echo "Kafka Quick Test"
echo "================"
echo ""

# Create topic
echo "1. Creating topic..."
docker exec kafka kafka-topics \
    --bootstrap-server $BROKER \
    --create \
    --topic $TOPIC \
    --partitions 3 \
    --replication-factor 1 \
    --if-not-exists 2>/dev/null

# Produce messages
echo "2. Producing 5 messages..."
for i in {1..5}; do
    echo "Test message $i" | docker exec -i kafka kafka-console-producer \
        --bootstrap-server $BROKER \
        --topic $TOPIC 2>/dev/null
done

# Consume messages
echo "3. Consuming messages..."
docker exec kafka kafka-console-consumer \
    --bootstrap-server $BROKER \
    --topic $TOPIC \
    --from-beginning \
    --max-messages 5 \
    --timeout-ms 5000 2>/dev/null

echo ""
echo "✓ Quick test completed successfully!"

# Cleanup
echo ""
read -p "Delete test topic? [y/N] " cleanup
if [ "$cleanup" = "y" ]; then
    docker exec kafka kafka-topics \
        --bootstrap-server $BROKER \
        --delete \
        --topic $TOPIC 2>/dev/null
    echo "✓ Test topic deleted"
fi

## Troubleshooting

If scripts fail:
1. Check Docker is running: `docker ps`
2. Check Kafka is ready: `make health`
3. View logs: `make logs`
4. Restart: `make stop && make start-single`
